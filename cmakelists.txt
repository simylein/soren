cmake_minimum_required(VERSION 3.13)

set(PICO_SDK_PATH "${CMAKE_SOURCE_DIR}/pico-sdk")
set(PICO_EXTRAS_PATH "${CMAKE_SOURCE_DIR}/pico-extras")

include(${PICO_SDK_PATH}/external/pico_sdk_import.cmake)
include(${PICO_EXTRAS_PATH}/external/pico_extras_import.cmake)

project(soren_project)

pico_sdk_init()

add_executable(soren src/main.c src/rp2040.c src/pcf8563.c src/si7021.c src/sx1278.c)

pico_enable_stdio_usb(soren 1)
pico_enable_stdio_uart(soren 0)

target_link_libraries(soren pico_stdlib hardware_adc hardware_i2c hardware_spi hardware_sleep)

pico_add_extra_outputs(soren)
