cmake_minimum_required(VERSION 3.13)

set(PICO_SDK_PATH "${CMAKE_SOURCE_DIR}/pico-sdk")
set(PICO_EXTRAS_PATH "${CMAKE_SOURCE_DIR}/pico-extras")

include(${PICO_SDK_PATH}/external/pico_sdk_import.cmake)
include(${PICO_EXTRAS_PATH}/external/pico_extras_import.cmake)

project(luna_project)

pico_sdk_init()

add_executable(luna main.c)

pico_enable_stdio_usb(luna 1)
pico_enable_stdio_uart(luna 0)

target_link_libraries(luna pico_stdlib hardware_adc hardware_sleep)

pico_add_extra_outputs(luna)
